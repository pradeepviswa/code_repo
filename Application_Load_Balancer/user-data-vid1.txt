#!/bin/bash
# Update and install Apache
yum update -y
yum install -y httpd

# Enable and start Apache service
systemctl enable httpd
systemctl start httpd

# Create directories
mkdir -p /var/www/html/vid


# Create first test page (red background)
cat <<EOF > /var/www/html/vid/index.html
<html>
  <head><title>VID Server 1</title></head>
  <body style="background-color:red; color:white; text-align:center;">
    <h1>Welcome to VID Server 1</h1>
    <p>This page is served from /var/www/html/vid</p>
  </body>
</html>
EOF

# Restart Apache to ensure content is served
systemctl restart httpd
